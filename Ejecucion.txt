# Actualizar el sistema
sudo apt update && sudo apt upgrade -y

# Instalar Node.js y npm (para el Frontend)
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt install -y nodejs

# Instalar Python y pip (para el Backend)
sudo apt install -y python3 python3-pip python3-venv

# Verificar instalaciones
node -v
npm -v
python3 --version



# Ejecucion del backend:
# Entrar a la carpeta del backend
cd backend

# Crear un entorno virtual (Opcional pero recomendado)
python3 -m venv venv
source venv/bin/activate

# Instalar dependencias
pip install -r requirements.txt

# Ejecutar el servidor
# Opción A: Usando el script que ya tienes (darle permisos primero)
chmod +x run.sh
./run.sh

# Opción B: Comando manual
uvicorn main:app --host 0.0.0.0 --port 8000 --reload


#Ejecucion del frontend:
# Regresar a la raíz del proyecto
cd ..

# Instalar las dependencias de Node
npm install

# Instalar Angular CLI de forma global (opcional, para usar el comando 'ng')
sudo npm install -g @angular/cli

# Ejecutar el frontend
npm start
# o también: ng serve --host 0.0.0.0
